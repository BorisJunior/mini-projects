def remove_outliers_using_iqr(df, column):
    """
    On va retitrer les valeurs aberrantes en utilisant la méthode IQR
    """
    # Calcul IQR
    Q1 = df[column].quantile(0.25)
    Q3 = df[column].quantile(0.75)
    IQR = Q3 - Q1
    
    # Definition des limites
    lower_bound = Q1 - 0.5 * IQR
    upper_bound = Q3 + 0.5 * IQR
    
    # Filter the data
    return df[(df[column] >= lower_bound) & (df[column] <= upper_bound)]

# Remove outliers for Quantity and UnitPrice
data_cleaned = remove_outliers_using_iqr(ecommerce_data_cleaned, 'Quantity')
data_cleaned = remove_outliers_using_iqr(ecommerce_data_cleaned, 'UnitPrice')
data_cleaned = remove_outliers_using_iqr(ecommerce_data_cleaned, 'TotalSales')

from sklearn.cluster import KMeans

# Calcul de l'inertie pour différentes valeurs de K
inertia = []
range_values = range(1, 15)

for k in range_values:
    kmeans = KMeans(n_clusters=k, random_state=42)
    kmeans.fit(scaled_data)
    inertia.append(kmeans.inertia_)

# Tracé de la méthode du coude
plt.figure(figsize=(10,6))
plt.plot(range_values, inertia, marker='o', linestyle='--')
plt.title('Méthode du coude')
plt.xlabel('Nombre de clusters')
plt.ylabel('Inertie')
plt.grid(True)
plt.show()
